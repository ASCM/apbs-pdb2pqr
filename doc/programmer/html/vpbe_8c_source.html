482 VPUBLIC double Vpbe_getCoulombEnergy1(Vpbe *thee) {
483 
484  int i, j, k, natoms;
485 
486  double dist, *ipos, *jpos, icharge, jcharge;
487  double energy = 0.0;
488  double eps, T;
489  Vatom *iatom, *jatom;
490  Valist *alist;
491 
492  VASSERT(thee != VNULL);
493  alist = Vpbe_getValist(thee);
494  VASSERT(alist != VNULL);
495  natoms = Valist_getNumberAtoms(alist);
496 
497  /* Do the sum */
498  for (i=0; i<natoms; i++) {
499  iatom = Valist_getAtom(alist,i);
500  icharge = Vatom_getCharge(iatom);
501  ipos = Vatom_getPosition(iatom);
502  for (j=i+1; j<natoms; j++) {
503  jatom = Valist_getAtom(alist,j);
504  jcharge = Vatom_getCharge(jatom);
505  jpos = Vatom_getPosition(jatom);
506  dist = 0;
507  for (k=0; k<3; k++) dist += ((ipos[k]-jpos[k])*(ipos[k]-jpos[k]));
508  dist = VSQRT(dist);
509  energy = energy + icharge*jcharge/dist;
510  }
511  }
512 
513  /* Convert the result to J */
514  T = Vpbe_getTemperature(thee);
515  eps = Vpbe_getSoluteDiel(thee);
516  energy = energy*Vunit_ec*Vunit_ec/(4*Vunit_pi*Vunit_eps0*eps*(1.0e-10));
517 
518  /* Scale by Boltzmann energy */
519  energy = energy/(Vunit_kb*T);
520 
521  return energy;
522 }
