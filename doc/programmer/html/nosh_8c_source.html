00960

00963

00973                 return 0;

00976

00979                 (Vstring_isdigit(tok) == 1)) {
00980                 if (expect == 0) {
00981                     thee->printcalc[idx][thee->printnarg[idx]] = ti-1;
00982                     expect = 1;
00983                 } else {
00984                     Vnm_print(2, "NOsh_parsePRINT:  Syntax error in PRINT \
00985 section while reading %s!\n", tok);
00986                     return 0;

00990                 if (expect == 1) {
00991                     thee->printop[idx][thee->printnarg[idx]] = 0;

00999                     }
01000                 } else {
01001                     Vnm_print(2, "NOsh_parsePRINT:  Syntax error in PRINT \
01002 section while reading %s!\n", tok);
01003                     return 0;
01004                 }
01005                 /* Grab subtraction operation */
01006             } else if (Vstring_strcasecmp(tok, "-") == 0) {
01007                 if (expect == 1) {
01008                     thee->printop[idx][thee->printnarg[idx]] = 1;
01009                     (thee->printnarg[idx])++;
01010                     expect = 0;
01011                     if (thee->printnarg[idx] >= NOSH_MAXPOP) {
01012                         Vnm_print(2, "NOsh_parseREAD:  Exceeded max number \

01016                     }

01018                     Vnm_print(2, "NOsh_parsePRINT:  Syntax error in PRINT \
01019 section while reading %s!\n", tok);
01020                     return 0;
01021                 }
01022                 /* Grab a calculation name from elec ID */
01023             } else if (sscanf(tok, "%s", name) == 1) {
01024                 if (expect == 0) {
01025                     for (ielec=0; ielec<thee->nelec; ielec++) {
01026                         if (Vstring_strcasecmp(thee->elecname[ielec], name) == 0) {

01043                     }

01045                     Vnm_print(2, "NOsh_parsePRINT:  Syntax error in PRINT \

01047                     return 0;

01803 
01804         /* Only the finest level handles I/O and needs to worry about disjoint
01805             partitioning */
01806         if (ifocus != (nfocus-1)) calcf->pbeparm->numwrite = 0;

01814                 calcf->mgparm->partDisjLength[j] = calcf->mgparm->glen[j];
01815             }
01816         }
01817 
01818 
01819         calcf->mgparm->parsed = 1;
01820     }
