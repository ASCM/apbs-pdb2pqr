412 VPUBLIC int NOsh_parseInput(
413  NOsh *thee,
414  Vio *sock
415  ) {
416 
417  char *MCwhiteChars = " =,;\t\r\n";
418  char *MCcommChars = "#%";
419  char tok[VMAX_BUFSIZE];
420 
421  if (thee == VNULL) {
422  Vnm_print(2, "NOsh_parseInput: Got NULL thee!\n");
423  return 0;
424  }
425 
426  if (sock == VNULL) {
427  Vnm_print(2, "NOsh_parseInput: Got pointer to NULL socket!\n");
428  Vnm_print(2, "NOsh_parseInput: The specified input file was not found!\n");
429  return 0;
430  }
431 
432  if (thee->parsed) {
433  Vnm_print(2, "NOsh_parseInput: Already parsed an input file!\n");
434  return 0;
435  }
436 
437  if (Vio_accept(sock, 0) < 0) {
438  Vnm_print(2, "NOsh_parseInput: Problem reading from socket!\n");
439  return 0;
440  }
441 
442  /* Set up the whitespace and comment character definitions */
443  Vio_setWhiteChars(sock, MCwhiteChars);
444  Vio_setCommChars(sock, MCcommChars);
445 
446  /* We parse the file until we run out of tokens */
447  Vnm_print(0, "NOsh_parseInput: Starting file parsing...\n");
448  while (Vio_scanf(sock, "%s", tok) == 1) {
449  /* At the highest level, we look for keywords that indicate functions like:
450 
451  read => Read in a molecule file
452  elec => Do an electrostatics calculation
453  print => Print some results
454  apolar => do a non-polar calculation
455  quit => Quit
456 
457  These cause the code to go to a lower-level parser routine which
458  handles keywords specific to the particular function. Each
459  lower-level parser routine then returns when it hits the "end"
460  keyword. Due to this simple layout, no nesting of these "function"
461  sections is allowed.
462  */
463  if (Vstring_strcasecmp(tok, "read") == 0) {
464  Vnm_print(0, "NOsh: Parsing READ section\n");
465  if (!NOsh_parseREAD(thee, sock)) return 0;
466  Vnm_print(0, "NOsh: Done parsing READ section \
467 (nmol=%d, ndiel=%d, nkappa=%d, ncharge=%d, npot=%d)\n", thee->nmol, thee->ndiel,
468  thee->nkappa, thee->ncharge,thee->npot);
469  } else if (Vstring_strcasecmp(tok, "print") == 0) {
470  Vnm_print(0, "NOsh: Parsing PRINT section\n");
471  if (!NOsh_parsePRINT(thee, sock)) return 0;
472  Vnm_print(0, "NOsh: Done parsing PRINT section\n");
473  } else if (Vstring_strcasecmp(tok, "elec") == 0) {
474  Vnm_print(0, "NOsh: Parsing ELEC section\n");
475  if (!NOsh_parseELEC(thee, sock)) return 0;
476  Vnm_print(0, "NOsh: Done parsing ELEC section (nelec = %d)\n",
477  thee->nelec);
478  } else if (Vstring_strcasecmp(tok, "apolar") == 0) {
479  Vnm_print(0, "NOsh: Parsing APOLAR section\n");
480  if (!NOsh_parseAPOLAR(thee, sock)) return 0;
481  Vnm_print(0, "NOsh: Done parsing APOLAR section (nelec = %d)\n",
482  thee->nelec);
483  } else if (Vstring_strcasecmp(tok, "quit") == 0) {
484  Vnm_print(0, "NOsh: Done parsing file (got QUIT)\n");
485  break;
486  } else {
487  Vnm_print(2, "NOsh_parseInput: Ignoring undefined keyword %s!\n", tok);
488  }
489  }
490 
491  thee->parsed = 1;
492  return 1;
493
