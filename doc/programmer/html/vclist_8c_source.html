130 VPRIVATE Vrc_Codes Vclist_setupGrid(Vclist *thee) {
131 
132  /* Inflation factor ~ sqrt(2)*/
133  #define VCLIST_INFLATE 1.42
134 
135  int i;
136  double length[VAPBS_DIM], r_max;
137 
138  /* Set up the grid corners */
139  switch (thee->mode) {
140  case CLIST_AUTO_DOMAIN:
141  /* Get molecule dimensions */
142  Vclist_getMolDims(thee, thee->lower_corner, thee->upper_corner,
143  &r_max);
144  /* Set up grid spacings */
145  for (i=0; i<VAPBS_DIM; i++) {
146  thee->upper_corner[i] = thee->upper_corner[i]
147  + VCLIST_INFLATE*(r_max+thee->max_radius);
148  thee->lower_corner[i] = thee->lower_corner[i]
149  - VCLIST_INFLATE*(r_max+thee->max_radius);
150  }
151  break;
152  case CLIST_MANUAL_DOMAIN:
153  /* Grid corners established in constructor */
154  break;
155  default:
156  Vnm_print(2, "Vclist_setupGrid: invalid setup mode (%d)!\n",
157  thee->mode);
158  return VRC_FAILURE;
159  }
160 
161  /* Set up the grid lengths and spacings */
162  for (i=0; i<VAPBS_DIM; i++) {
163  length[i] = thee->upper_corner[i] - thee->lower_corner[i];
164  thee->spacs[i] = length[i]/((double)(thee->npts[i] - 1));
165  }

170  (thee->lower_corner)[0], (thee->lower_corner)[1],
171  (thee->lower_corner)[2]);
172 
173  return VRC_SUCCESS;

236 VPRIVATE void Vclist_gridSpan(Vclist *thee,
237  Vatom *atom, /* Atom */
238  int imin[VAPBS_DIM], /* Set to min grid indices */
239  int imax[VAPBS_DIM] /* Set to max grid indices */
240  ) {
241 
242  int i;
243  double *coord, dc, idc, rtot;
244 
245  /* Get the position in the grid's frame of reference */
246  coord = Vatom_getPosition(atom);
247 
248  /* Get the range the atom radius + probe radius spans */
249  rtot = Vatom_getRadius(atom) + thee->max_radius;
250 
251  /* Calculate the range of grid points the inflated atom spans in the x
252  * direction. */
253  for (i=0; i<VAPBS_DIM; i++) {
254  dc = coord[i] - (thee->lower_corner)[i];
255  idc = (dc + rtot)/(thee->spacs[i]);
256  imax[i] = (int)(ceil(idc));
257  imax[i] = VMIN2(imax[i], thee->npts[i]-1);
258  idc = (dc - rtot)/(thee->spacs[i]);
259  imin[i] = (int)(floor(idc));
260  imin[i] = VMAX2(imin[i], 0);
261  }
