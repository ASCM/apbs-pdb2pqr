116 VPUBLIC int Vopot_pot(Vopot *thee, double pt[3], double *value) {

121  Valist *alist;
122

124

126  xkappa = (1.0e10)*Vpbe_getXkappa(thee->pbe);
127  T = Vpbe_getTemperature(thee->pbe);
128  alist = Vpbe_getValist(thee->pbe);
129 
130  u = 0;
131 
132  /* See if we're on the mesh */
133  if (Vmgrid_value(thee->mgrid, pt, &u)) {
134 
135  *value = u;
136 
137  } else {
138 
139  switch (thee->bcfl) {
140 
141  case BCFL_ZERO:
142  u = 0;
143  break;
144 
145  case BCFL_SDH:
146  size = (1.0e-10)*Vpbe_getSoluteRadius(thee->pbe);
147  position = Vpbe_getSoluteCenter(thee->pbe);
148  charge = Vunit_ec*Vpbe_getSoluteCharge(thee->pbe);
149  dist = 0;
150  for (i=0; i<3; i++)
151  dist += VSQR(position[i] - pt[i]);
152  dist = (1.0e-10)*VSQRT(dist);
153  val = (charge)/(4*VPI*Vunit_eps0*eps_w*dist);
154  if (xkappa != 0.0)

158  break;
